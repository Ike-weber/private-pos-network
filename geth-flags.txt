  --ws.addr 127.0.0.1           bind WS to localhost only
  --authrpc.addr 127.0.0.1      bind engine API to localhost only
  --http.vhosts 'localhost'     restrict Host header (TLS handled by reverse proxy)
  --http.corsdomain 'localhost'   restrict CORS origin
